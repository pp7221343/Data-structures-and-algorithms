package com.wms.demo.struct;

public class MyLinkedList {

    private Node head;

    public class Node{
        private int data;
        private Node next;
        Node(int a){
            this.data = a;
        }
    }

    public void add(int a){
        Node node = new Node(a);
        if (head != null) {
            Node temp = head;
            while (temp.next != null) {
                temp = temp.next;
            }
            temp.next = node;
        } else{
            head = node;
        }
        node.next = null;
    }


    //获取倒数第k个节点
    public int get(int k){
        if(k == 0){
            return 0;
        }
        int j=k;
        Node fast = head;
        Node slow = head;
        for(int i = 0;i<k-1;i++){
            if(fast.next!=null){
                fast = fast.next;
            }
        }
        while(fast.next!=null){
            fast = fast.next;
            slow = slow.next;
        }

        return slow.data;
    }
}
